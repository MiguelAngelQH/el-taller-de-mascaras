Prueba de aceptación ATDD-05 a ATDD-10 — GDScript con GUT
extends GutTest

# ATDD-07: el inventario se abre y cierra sin error
func test_inventario_toggle():
    var hub = preload("res://Objects/assets/inventoryHub.tscn").instantiate()
    add_child(hub)
    # Estado inicial cerrado
    assert_false(hub.isShowInventory)
    # Simular apertura
    hub.isShowInventory = true
    hub.cambiar_visibilidad()
    assert_true(Global.isInventary)
    # Simular cierre
    hub.isShowInventory = false
    hub.cambiar_visibilidad()
    assert_false(Global.isInventary)
    hub.queue_free()

# ATDD-08: inventario tiene al menos 6 slots
func test_inventario_tiene_slots_suficientes():
    var hub = preload("res://Objects/assets/inventoryHub.tscn").instantiate()
    add_child(hub)
    var grid = hub.find_child("InventoryGridContainer", true, false)
    assert_not_null(grid)
    assert_gte(grid.get_child_count(), 6)
    hub.queue_free()

# ATDD-09: swap de ítems entre slots no pierde datos
func test_swap_no_pierde_item():
    var a = ItemData.new(); a.item_name = "PotionA"
    var b = ItemData.new(); b.item_name = "PotionB"
    var tmp = a; a = b; b = tmp
    assert_eq(a.item_name, "PotionB")
    assert_eq(b.item_name, "PotionA")
